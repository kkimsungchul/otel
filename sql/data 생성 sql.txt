package com.sungchul.macro;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Random;

public class MainThree {
    public static void main(String[]args){
        int leftLimit = 48; // numeral '0'
        int rightLimit = 122; // letter 'z'
        int titleLength = 20;
        int contentLength = 300;
        Random random = new Random();


        String fileName = "data.sql";
        String fileContent = "INSERT INTO board (TITLE, CONTENT,CREATE_DATE,UPDATE_DATE) VALUES";
        // File 객체를 생성
        File file = new File(fileName);
        //UUID uuid=null;
        try {
            FileWriter fw = new FileWriter(file);
            BufferedWriter bw = new BufferedWriter(fw);
            String title;
            String content;
            bw.write(fileContent);
            for(int i=0;i<10000;i++){
                title = random.ints(leftLimit,rightLimit + 1)
                        .filter(j -> (j <= 57 || j >= 65) && (j <= 90 || j >= 97))
                        .limit(titleLength)
                        .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
                        .toString();
                content = random.ints(leftLimit,rightLimit + 1)
                        .filter(j -> (j <= 57 || j >= 65) && (j <= 90 || j >= 97))
                        .limit(contentLength)
                        .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
                        .toString();
                fileContent = ",('"+title+"', '"+content+"',CURRENT_TIMESTAMP() ,CURRENT_TIMESTAMP())";
                bw.write(fileContent);
            }

            bw.close();

            System.out.println("파일에 내용이 성공적으로 작성되었습니다.");
        } catch (IOException e) {
            System.out.println("파일에 내용을 작성하는 동안 오류가 발생했습니다.");
            e.printStackTrace();
        }

    }
}
